The probe loop body (transcribed for the future reader; the
scratch script `.tmp_t008_probe.sh` is removed at session close):

```bash
# Scenario 1
grep -E '^#{1,6} Model pinning' README.md

# Scenario 2 — extract pinning section, then grep every named row
pin_section=$(awk '/^## Model pinning/{flag=1;next} flag && /^## /{flag=0} flag' README.md)
for phase in speccy-tasks speccy-work speccy-ship speccy-init speccy-review; do
  grep -qF "$phase" <(echo "$pin_section")
done
for persona in reviewer-business reviewer-tests reviewer-architecture \
               reviewer-security reviewer-style reviewer-docs; do
  grep -qF "$persona" <(echo "$pin_section")
done
grep -qF 'sonnet[1m]' <(echo "$pin_section")
grep -qF 'opus[1m]'   <(echo "$pin_section")
grep -qF 'gpt-5.5'    <(echo "$pin_section")
grep -qi 'unpinned'   <(echo "$pin_section")

# Scenario 3
grep -qE '/agent speccy-'      <(echo "$pin_section")
grep -qiE 'parent session'     <(echo "$pin_section")

# Scenario 4
grep -qiE 'edit.*(frontmatter|agent file)' <(echo "$pin_section")
grep -qiE 'alias'                          <(echo "$pin_section")

# Scenario 5
grep -qiE 'speccy-review.*(unpinned|TASKS\.md|orchestrator|REQ-009)' \
  <(echo "$pin_section")

# Scenario 6 — every `speccy <cmd>` mention outside the pinning section
# matches one of the ten shipped commands (init plan tasks implement review
# report status next check verify); `speccy speccy` is the cargo install
# package-name token and excluded as such.
rest=$(awk '/^## Model pinning/{flag=1;next} flag && /^## /{flag=0} !flag' README.md)
echo "$rest" | grep -oE 'speccy [a-z][a-z-]+' | sort -u | awk '
  $2 != "init" && $2 != "plan" && $2 != "tasks" && $2 != "implement" &&
  $2 != "review" && $2 != "report" && $2 != "status" && $2 != "next" &&
  $2 != "check" && $2 != "verify" && $2 != "speccy" { print }
'

# Scenario 7
grep -nE '\.speccy/skills/' README.md
```
